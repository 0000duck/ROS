386  static RCS::Rotation Convert2Rotation(IkReal *eerot) {
387  double q0 = (eerot[0] + eerot[4] + eerot[8] + 1.0f) / 4.0f;
388  double q1 = (eerot[0] - eerot[4] - eerot[8] + 1.0f) / 4.0f;
389  double q2 = (-eerot[0] + eerot[4] - eerot[8] + 1.0f) / 4.0f;
390  double q3 = (-eerot[0] - eerot[4] + eerot[8] + 1.0f) / 4.0f;
391 
392  if (q0 < 0.0f) {
393  q0 = 0.0f;
394  }
395 
396  if (q1 < 0.0f) {
397  q1 = 0.0f;
398  }
399 
400  if (q2 < 0.0f) {
401  q2 = 0.0f;
402  }
403 
404  if (q3 < 0.0f) {
405  q3 = 0.0f;
406  }
407  q0 = sqrt(q0);
408  q1 = sqrt(q1);
409  q2 = sqrt(q2);
410  q3 = sqrt(q3);
411 
412  if ((q0 >= q1) && (q0 >= q2) && (q0 >= q3)) {
413  q0 *= +1.0f;
414  q1 *= SIGN(eerot[7] - eerot[5]);
415  q2 *= SIGN(eerot[2] - eerot[6]);
416  q3 *= SIGN(eerot[3] - eerot[1]);
417  } else if ((q1 >= q0) && (q1 >= q2) && (q1 >= q3)) {
418  q0 *= SIGN(eerot[7] - eerot[5]);
419  q1 *= +1.0f;
420  q2 *= SIGN(eerot[3] + eerot[1]);
421  q3 *= SIGN(eerot[2] + eerot[6]);
422  } else if ((q2 >= q0) && (q2 >= q1) && (q2 >= q3)) {
423  q0 *= SIGN(eerot[2] - eerot[6]);
424  q1 *= SIGN(eerot[3] + eerot[1]);
425  q2 *= +1.0f;
426  q3 *= SIGN(eerot[7] + eerot[5]);
427  } else if ((q3 >= q0) && (q3 >= q1) && (q3 >= q2)) {
428  q0 *= SIGN(eerot[3] - eerot[1]);
429  q1 *= SIGN(eerot[6] + eerot[2]);
430  q2 *= SIGN(eerot[7] + eerot[5]);
431  q3 *= +1.0f;
432  } else {
433  throw std::runtime_error("Error while converting to quaternion! \n");
434  }
435  double r = NORM(q0, q1, q2, q3);
436  q0 /= r;
437  q1 /= r;
438  q2 /= r;
439  q3 /= r;
440  RCS::Rotation q(q0, q1, q2, q3);
441  return q;
442  }
443  // Convert input effector pose, in w x y z quaternion notation, to rotation matrix.
444  // Must use doubles, else lose precision compared to directly inputting the rotation matrix.
445  // Found at http://kaist-ros-pkg.googlecode.com/svn/trunk/arm_kinematics_tools/src/ikfastdemo/ikfastdemo.cpp
446 
447  static void Convert2RotationMatrix(const RCS::Rotation & quat, IkReal *eerot) {
448  double qq1 = 2 * quat.x() * quat.x();
449  double qq2 = 2 * quat.y() * quat.y();
450  double qq3 = 2 * quat.z() * quat.z();
451  eerot[3 * 0 + 0] = 1 - qq2 - qq3;
452  eerot[3 * 0 + 1] = 2 * (quat.x() * quat.y() - quat.w() * quat.z());
453  eerot[3 * 0 + 2] = 2 * (quat.x() * quat.z() + quat.w() * quat.y());
454  eerot[3 * 1 + 0] = 2 * (quat.x() * quat.y() + quat.w() * quat.z());
455  eerot[3 * 1 + 1] = 1 - qq1 - qq3;
456  eerot[3 * 1 + 2] = 2 * (quat.y() * quat.z() - quat.w() * quat.x());
457  eerot[3 * 2 + 0] = 2 * (quat.x() * quat.z() - quat.w() * quat.y());
458  eerot[3 * 2 + 1] = 2 * (quat.y() * quat.z() + quat.w() * quat.x());
459  eerot[3 * 2 + 2] = 1 - qq1 - qq2;
460  }
