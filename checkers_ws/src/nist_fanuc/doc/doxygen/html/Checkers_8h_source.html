332 
333  for (size_t n = 0; n < tries; n++) {
334  nJump = 1;
335  if (n == 1) nDir = -1 * nDir; //second time for kings
336

338  && LegalRow(row + nDir * nJump)

340  moves.push_back(Move(row + nDir * nJump, col + nJump));
341 
342  if (!bJumpOnly && (col - nJump >= 0)
343  && LegalRow(row + nDir * nJump)
344  && inboard[row + nDir * nJump][col - nJump] == EMPTY)
345  moves.push_back(Move(row + nDir * nJump, col - nJump));
346 
347  nJump = 2;
348  if ((col + nJump < 8) && LegalRow(row + nDir * nJump) &&
349  inboard[row + nDir * nJump][col + nJump] == EMPTY &&
350  inboard[row + nDir * (nJump - 1)][col + nJump - 1] == opponent) {
351  Move move(row + nDir*nJump, col + nJump, true);
352  BoardType newboard = MakeMove(inboard, player, row, col, move);
353  move.doublejumps = BuildMoves(newboard, player, move, true);

358  inboard[row + nDir * nJump][col - nJump] == EMPTY &&
359  inboard[row + nDir * (nJump - 1)][col - nJump + 1] == opponent) {
