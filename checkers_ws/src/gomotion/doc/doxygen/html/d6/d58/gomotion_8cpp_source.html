352  Given:
353 
354  The current queue, 'queue'.
355 
356  The joint motion to append, 'motion', with these set:
357 
358  motion->end.u.joint[] for the target joint positions in abs coords;
359  motion->profile for the type (CV, CA, CJ);
360  motion->par[] for the joint v,a,j.
361 
362  Computed:
363 
364  The motion spec to be appended, 'spec', copied from *motion and then
365  filled in as follows:
366 
367  spec.start.u.joint[] for the start positions (from this->there.u.joint[]);
368  spec.cv,a,j[] for the phase distances and times for simultaneous arrival;
369  spec.totalt for the max time.
370 
371  The queue is modified as follows:
372 
373  this->there.u.joint[] for the end positions (from spec.end.u.joint[]);
374  the queue pointers for an append.
375  */

377  go_result
378  go_motion_queue::append_joint(
379  const go_motion_spec * motion)
380  {
381  go_motion_spec spec;
382  go_real td, maxtd;
383  go_integer i;
384 
385  if (this->_number >= this->_size) {
386  return GO_RESULT_NO_SPACE;
387  }
388 
389  spec = *motion;
390  maxtd = spec.totalt; /* set the max to be the provided time,
391  which is 0 normally for speed-based
392  moves but will be non-zero for time-based
393  moves */
394 
395  /* fill in the motion specs (phase times, distances) for each joint,
396  keeping track of the maximum time. This maximum time will be used
397  to scale all the joint motions so that they arrive at the same time,
398  and will be the overall time 'totalt' for the motion. */
399 
400  for (i = 0; i < this->_joint_num; i++) {
401  /* set the start joint value to be the end of the previous move */
402  spec.start.u.joint[i] = this->_there.u.joint[i];
403 
404  /* 'td' is the incremental distance traveled by this joint */
405  td = fabs(spec.end.u.joint[i] - this->_there.u.joint[i]);
406 
407  /* set the end joint values for the queue */
408  this->_there.u.joint[i] = spec.end.u.joint[i];
409  go_traj_cj_compute(td, spec.par[i].vel, spec.par[i].acc, spec.par[i].jerk, &spec.cj[i]);
410  if (spec.cj[i].tend > maxtd) maxtd = spec.cj[i].tend;
411  }
412 
413  spec.totalt = maxtd; /* set the overall motion time */
414 
415  /* now all the joint motion phase times are calculated, and we have
416  the maximum time, so scale all the joint moves to arrive at the
417  same time */
418 
419  for (i = 0; i < this->_joint_num; i++) {
420  go_traj_cj_scale(&spec.cj[i], maxtd, &spec.cj[i]);
421  }
422 
423  /* copy the spec to the queue, and update the queue pointers */
424  *this->_end = spec;
425  this->_number++;
426  this->_end++;
427  if (this->_end >= this->_endptr) {
428  this->_end = this->_startptr;
429  }
430  this->_last_id = spec.id;
431 
432  return GO_RESULT_OK;
433  }

483  /*
484  Circular motion is done in cylindrical coordinates, enabling purely
485  circular (theta) motion as well as helical (z) motion and spiral (r)
486  motion. Motion is parameterized by curve length in the trajectory
487  planner, not in the natural polar angle. The curve length of a
488  general helical spiral is closed-form but not invertible, so we
489  can't get polar angle as a function of arc length exactly. We can
490  approximate arc length as the Cartesian sum of the average azimuthal
491  displacment and axial displacement. Given this, we can calculate an
492  approximate polar angle and a resulting trajectory point that lies
493  exactly on the curve and moves close to the desired speed.
494 
495  Given the start pose, end pose, center vector, normal vector and
496  number of turns, we compute some quantities that most efficiently
497  allow us to interpolate along the general curve with an arc length
498  parameterization.
499 
500  First we get the estimated total arc length as
501 
502  stot = sqrt(sq(average azimuthal distance) + sq(axial distance))
503 
504  where the average azimuthal distance is the distance swept out at
505  the average of the starting- and ending radii as the angle moves
506  from the starting radial vector to the ending radial vector
507  (cylindrical theta coordinate); and the axial distance is the
508  distance along the helical axis (cylindrical z coordinate).
509 
510  This is a close approximation. For example, for a one-turn spiral
511  starting at r = 1 and ending at r = 2, the exact arc length is 9.48
512  and the approximate arc length is 9.42.
513 
514  Let 'normal' be the unit normal vector, and 'qnormal' be 'normal'
515  considered as a rotation vector converted to a quaternion. 'normal'
516  will be used to slide along the Z axis. 'qnormal' will be used to
517  sweep the radial vector.
518 
519  Let 'rcs' be the vector from the center to the start, 'rcsp' be the
520  projection of 'rcs' into the normal plane with length 'rstart', and
521  'zstart' be the vector from the plane to 'rcs'.
522 
523  Likewise, let 'rce' be the vector from the center to the end, 'rcep'
524  be the projection of 'rce' into the normal plane with length 'rend',
525  and 'zend' be the vector from the plane to 'rce'.
526 
527  Let 'rtot' = 'rend' - 'rstart' be the signed difference between the
528  ending and starting radii. Let 'zvec' be the vector from 'zstart' to
529  'zend', with 'zuvec' and 'zmag' the unit vector and magnitude,
530  respectively.
531 
532  Let 'urcsp' be the unit vector along 'rcsp'. This is the unit vector
533  that initially points toward the projection of the starting pose
534  onto the normal plane, and sweeps around as motion proceeds, always
535  pointing toward the projection of the trajectory point onto the
536  normal plane.
537 
538  Let 'thtot' be the total angle of rotation that brings 'rcsp' to
539  'rcep', including any extra turns. Let 'srtot' be the azimuthal
540  length swept out by the average radial vector through 'thtot'. Let
541  'ztot' be the distance from 'zstart' to 'zend'. Let 'stot' be the
542  Cartesian sum of 'srtot' and 'ztot'. Let 'stotinv' be its inverse.
543 
544  Given a length along the curve 's' calculated by the trajectory
545  planner, we multiply by 'stotinv' to get the fraction of the path
546  traversed, 'sfrac'. Calculate the fractions of the swept angle,
547  radial difference and z vectors as respectively
548 
549  thfrac = sfrac * thtot
550  rfrac = sfrac * rtot
551  zfrac = sfrac * ztot
552 
553  Scale the unit rotation quaternion 'qnormal' by 'thfrac' and rotate
554  the unit radial vector 'urcsp' by this. Scale by 'rstart' + 'rfrac' to
555  get the radial vector 'rvec'.
556 
557  Scale the unit normal vector 'normal' by 'zfrac' and add to 'zstart'
558  to bget the axial vector 'zvec'.
559 
560  Calculate the complete vector to the trajectory point as 'center' +
561  'rvec' + 'zvec'.
562 
563  The rotation from the starting orientation to the ending orientation
564  is handled as all motions, in which the "length" is the
565  magnitude of the rotation vector from starting orientation to the
566  ending orientation, which is scaled from zero to full magnitude
567  as the motion proceeds:
568 
569  rotend = rotstart * rot -> rot = inv(rotstart) * rotend
570 
571  Unitize 'rot' to get 'urot'.
572 
573  During interpolation, scale 'urot' by the trajectory planner's
574  current rotational distance, and get the current rotation as
575 
576  rotcur = rotstart * urot * rotational distance
577  */
578 
579  go_result
580  go_motion_queue::append_world(
581  const go_motion_spec * motion)
582  {
583  go_motion_spec spec;
584  go_quat quat;
585  go_cart cart;
586  go_cart rcs; /* vector from center to start */
587  go_cart rcsp; /* its projection onto normal plane */
588  go_cart rce; /* vector from center to end */
589  go_cart rcep; /* its projection onto normal plane */
590  go_real r;
591  go_real dot;
592  go_real ravg;
593  go_real time;
594  go_rvec rvec;
595  go_real td;
596  go_real rd;
597  go_result retval;
598 
599  if (this->_number >= this->_size) {
600  return GO_RESULT_NO_SPACE;
601  }
602 
603  spec = *motion;
604  time = spec.totalt; /* save requested time and override totalt
605  later if applicable */
606 
607  switch (spec.type) {
608  case GO_MOTION_LINEAR:
609  go_cart_cart_sub(&spec.end.u.pose.tran, &this->_there.u.pose.tran, &cart);
610  go_cart_mag(&cart, &td);
611  if (GO_RESULT_OK != go_cart_unit(&cart, &spec.u.lpar.uvec)) {
612  /* no translational motion, so set uvec to X */
613  spec.u.lpar.uvec.x = 1, spec.u.lpar.uvec.y = spec.u.lpar.uvec.z = 0;
614  }
615  break;
616 
617  case GO_MOTION_CIRCULAR:
618  /* unitize the normal vector and express it also as a unit rotation */
619  if (GO_RESULT_OK != go_cart_unit(&spec.u.cpar.normal, &spec.u.cpar.normal)) return GO_RESULT_ERROR;
620  (void) go_cart_rvec_convert(&spec.u.cpar.normal, &rvec);
621  (void) go_rvec_quat_convert(&rvec, &spec.u.cpar.qnormal);
622 
623  /* get the vector from the center to the start and project onto
624  the normal plane */
625  (void) go_cart_cart_sub(&this->_there.u.pose.tran, &spec.u.cpar.center, &rcs);
626  (void) go_cart_plane_proj(&rcs, &spec.u.cpar.normal, &rcsp);
627  /* save the projected starting radius */
628  (void) go_cart_mag(&rcsp, &spec.u.cpar.rstart);
629  /* save the vector from the normal plane to the start */
630  (void) go_cart_cart_sub(&rcs, &rcsp, &spec.u.cpar.zstart);
631  /* save the unit projected start vector for scaling and sweeping later */
632  if (GO_RESULT_OK != go_cart_unit(&rcsp, &spec.u.cpar.urcsp)) return GO_RESULT_ERROR;
633 
634  /* get the vector from the center to the end and project onto
635  the normal plane */
636  (void) go_cart_cart_sub(&spec.end.u.pose.tran, &spec.u.cpar.center, &rce);
637  (void) go_cart_plane_proj(&rce, &spec.u.cpar.normal, &rcep);
638  /* get the projected ending radius */
639  (void) go_cart_mag(&rcep, &r);
640  /* get the average radius */
641  ravg = 0.5 * (spec.u.cpar.rstart + r);
642  /* save the signed displacement from start to end radii */
643  spec.u.cpar.rtot = r - spec.u.cpar.rstart;
644  /* get the vector from the normal plane to the end */
645  (void) go_cart_cart_sub(&rce, &rcep, &cart);
646  /* and the vector from the start to the end */
647  (void) go_cart_cart_sub(&cart, &spec.u.cpar.zstart, &cart);
648  /* save the distance between z start and z end */
649  go_cart_mag(&cart, &spec.u.cpar.ztot);
650  /* we will use the normal vector to move from z start to z end,
651  so we need to make the total displacement negative if
652  zdiff and normal are antiparallel */
653  (void) go_cart_cart_dot(&cart, &spec.u.cpar.normal, &dot);
654  if (dot < 0.0) {
655  spec.u.cpar.ztot = -spec.u.cpar.ztot;
656  }
657 
658  /* compute angle between the projections. This will be the angle
659  we will interpolate */
660  go_cart_cart_angle(&rcsp, &rcep, &r);
661  /* now 'r' is the angle between the start and end vector projections,
662  in range [0..PI]. We need to correct this for more than half circles,
663  plus number of turns */
664  go_cart_cart_cross(&rcsp, &rcep, &cart);
665  go_cart_cart_dot(&cart, &spec.u.cpar.normal, &dot);
666  if (dot < 0.0) {
667  /* start X end lies in opposite direction of normal, so r should
668  be the larger one */
669  r = GO_2_PI - r;
670  }
671  /* now 'r' is a non-negative number, so we can add the number of turns */
672  r += GO_2_PI * spec.u.cpar.turns;
673  /* and save it out */
674  spec.u.cpar.thtot = r;
675 
676  /* get the average azimuthal distance */
677  r = ravg * spec.u.cpar.thtot;
678  /* get the Cartesian total distance */
679  r = sqrt(go_sq(r) + go_sq(spec.u.cpar.ztot));
680  td = r; /* we use td down below */
681  if (GO_TRAN_SMALL(r)) {
682  /* no translation, so set stotinv to some negative and thus invalid
683  value to signify this */
684  spec.u.cpar.stotinv = -1.0;
685  } else {
686  spec.u.cpar.stotinv = 1.0 / r;
687  }
688  break;
689 
690  default:
691  return GO_RESULT_ERROR;
692  } /* switch (spec.type) */
693 
694  /* do rotation the same way for either linear or circular moves */
695  if (GO_RESULT_OK != go_quat_inv(&this->_there.u.pose.rot, &quat)) {
696  /* 'there' should always be invertible; if not it wasn't initialized
697  to I, so do it here */
698  quat = this->_there.u.pose.rot = go_quat_identity();
699  }
700  retval = go_quat_quat_mult(&quat, &spec.end.u.pose.rot, &quat);
701  if (GO_RESULT_OK != retval) {
702  /* quat is OK, so spec.u.pose.u.pose.rot must not be */
703  return retval;
704  }
705  go_quat_mag(&quat, &rd);
706  if (GO_RESULT_OK != go_quat_unit(&quat, &spec.uquat)) {
707  /* no rotational motion, so set uquat I */
708  spec.uquat = go_quat_identity();
709  }
710 
711  go_traj_cj_compute(td, spec.par[0].vel, spec.par[0].acc, spec.par[0].jerk,
712  &spec.cj[0]);
713  go_traj_cj_compute(rd, spec.par[1].vel, spec.par[1].acc, spec.par[1].jerk,
714  &spec.cj[1]);
715  if (spec.cj[0].tend > spec.cj[1].tend) {
716  /* translation takes longer, so scale rotation */
717  spec.totalt = spec.cj[0].tend;
718  go_traj_cj_scale(&spec.cj[1], spec.totalt, &spec.cj[1]);
719  } else {
720  /* rotation takes longer, so scale translation */
721  spec.totalt = spec.cj[1].tend;
722  go_traj_cj_scale(&spec.cj[0], spec.totalt, &spec.cj[0]);
723  }
724  /* now scale both trans and rot motions to take 'time', if requested */
725  if (time > spec.totalt) {
726  spec.totalt = time;
727  go_traj_cj_scale(&spec.cj[0], spec.totalt, &spec.cj[0]);
728  go_traj_cj_scale(&spec.cj[1], spec.totalt, &spec.cj[1]);
729  }
730 
731  spec.start = this->_there;
732  this->_there = spec.end;
733  *this->_end = spec;
734  this->_number++;
735  this->_end++;
736  if (this->_end >= this->_endptr) {
737  this->_end = this->_startptr;
738  }
739  this->_last_id = spec.id;
740 
741  return GO_RESULT_OK;
742  }

866  go_result
867  go_motion_queue::interp_world(go_motion_spec * motion, go_real time, go_pose * pose)
868  {
869  go_traj_interp_spec tis;
870  go_traj_interp_spec ris;
871  go_cart cart;
872  go_quat quat;
873  go_real frac;
874 
875  go_traj_cj_interp(&motion->cj[0], time, &tis);
876  go_traj_cj_interp(&motion->cj[1], time, &ris);
877 
878  switch (motion->type) {
879  case GO_MOTION_LINEAR:
880  go_cart_scale_mult(&motion->u.lpar.uvec, tis.d, &cart);
881  go_cart_cart_add(&motion->start.u.pose.tran, &cart, &pose->tran);
882  break;
883 
884  case GO_MOTION_CIRCULAR:
885  if (motion->u.cpar.stotinv < 0) {
886  /* no translational motion around the circle, so we're there */
887  pose->tran = motion->end.u.pose.tran;
888  break;
889  }
890  pose->tran = motion->u.cpar.center; /* start at center */
891  frac = tis.d * motion->u.cpar.stotinv; /* get fraction of move */
892  /* scale rotation around axis */
893  go_quat_scale_mult(&motion->u.cpar.qnormal, frac * motion->u.cpar.thtot, &quat);
894  /* grow the radial vector */
895  go_cart_scale_mult(&motion->u.cpar.urcsp, motion->u.cpar.rstart + frac * motion->u.cpar.rtot, &cart);
896  /* sweep the radial vector */
897  go_quat_cart_mult(&quat, &cart, &cart);
898  /* add it to our pose vector */
899  go_cart_cart_add(&pose->tran, &cart, &pose->tran);
900  /* grow the z vector */
901  go_cart_scale_mult(&motion->u.cpar.normal, frac * motion->u.cpar.ztot, &cart);
902  /* add it to the starting z */
903  go_cart_cart_add(&motion->u.cpar.zstart, &cart, &cart);
904  /* add this to the pose */
905  go_cart_cart_add(&pose->tran, &cart, &pose->tran);
906  break;
907 
908  default:
909  return GO_RESULT_ERROR;
910  }
911 
912  /* rotation is handled the same way for either type */
913  go_quat_scale_mult(&motion->uquat, ris.d, &quat);
914  go_quat_quat_mult(&motion->start.u.pose.rot, &quat, &pose->rot);
915 
916  return GO_RESULT_OK;
917  }

975  go_result go_motion_queue::stop()
976  {
977  go_motion_spec * specptr ;
978  go_real now; /* time to begin stopping motion */
979  go_real endtime; /* time motion is really done */
980  go_integer i;
981  go_flag neg;
982  go_result retval;
983 
984  /* don't need to stop an empty or stopping queue */
985  if (0 == this->_number) {
986  return GO_RESULT_OK;
987  }
988 
989  /* drop all pending motions from the queue */
990  (void) this->drop_pending();
991 
992  /* point our attention at the current motion */
993  specptr = this->_start;
994 
995  /* we want to stop now, the current queue time */
996  now = this->_time;
997 
998  /* 'endtime' will the the revised stop time */
999  endtime = 0.0;
1000 
1001  /* Call go_traj_cj_stop for each joint- or tran/rot motion, so that
1002  each stops as fast as it can. The longest to stop will set the
1003  revised stop time, and each will be extended to stop at that
1004  longest time. Note that the specptr values are incremental for
1005  the move, as the original distance passed to go_traj_cj_compute
1006  was the incremental distance. */
1007 
1008  if (this->_type == GO_MOTION_JOINT) {
1009  for (i = 0; i < this->_joint_num; i++) {
1010  (void) go_traj_cj_stop(&specptr->cj[i], now, &specptr->cj[i]);
1011  maxit(specptr->cj[i].tend, endtime);
1012  }
1013  /* now 'endtime' is the longest stop time, so extend all the moves
1014  to that time, and revise the joint end positions and the
1015  queue end position 'there' */
1016  for (i = 0; i < this->_joint_num; i++) {
1017  neg = specptr->end.u.joint[i] < specptr->start.u.joint[i] ? 1 : 0;
1018  (void) go_traj_cj_extend(&specptr->cj[i], endtime, &specptr->cj[i]);
1019  if (neg) {
1020  specptr->end.u.joint[i] = specptr->start.u.joint[i] - specptr->cj[i].dtend;
1021  } else {
1022  specptr->end.u.joint[i] = specptr->start.u.joint[i] + specptr->cj[i].dtend;
1023  }
1024  /* revise the queue's end position */
1025  this->_there.u.joint[i] = specptr->end.u.joint[i];
1026  }
1027  } else if (this->_type == GO_MOTION_UJOINT) {
1028  for (i = 0; i < this->_joint_num; i++) {
1029  neg = specptr->end.u.joint[i] < specptr->start.u.joint[i] ? 1 : 0;
1030  (void) go_traj_cj_stop(&specptr->cj[i], now, &specptr->cj[i]);
1031  if (neg) {
1032  specptr->end.u.joint[i] = specptr->start.u.joint[i] - specptr->cj[i].dtend;
1033  } else {
1034  specptr->end.u.joint[i] = specptr->start.u.joint[i] + specptr->cj[i].dtend;
1035  }
1036  maxit(specptr->cj[i].tend, endtime);
1037  this->_there.u.joint[i] = specptr->end.u.joint[i];
1038  }
1039  } else if (this->_type == GO_MOTION_WORLD) {
1040  (void) go_traj_cj_stop(&specptr->cj[0], now, &specptr->cj[0]);
1041  maxit(specptr->cj[0].tend, endtime);
1042  (void) go_traj_cj_stop(&specptr->cj[1], now, &specptr->cj[1]);
1043  maxit(specptr->cj[1].tend, endtime);
1044  /* now 'endtime' is the longest stop time, so extend both tran and
1045  rot to that time */
1046  (void) go_traj_cj_extend(&specptr->cj[0], endtime, &specptr->cj[0]);
1047  (void) go_traj_cj_extend(&specptr->cj[1], endtime, &specptr->cj[1]);
1048  /* now the tran and rot are extended; now we need to recompute
1049  the new end translation and rotation */
1050  retval = interp_world(specptr, endtime, &specptr->end.u.pose);
1051  if (GO_RESULT_OK != retval) return retval;
1052  /* else revise the queue's end position */
1053  this->_there.u.pose = specptr->end.u.pose;
1054  } else {
1055  /* queue->type isn't JOINT, UJOINT or WORLD */
1056  return GO_RESULT_BUG;
1057  }
1058 
1059  specptr->totalt = endtime;
1060 
1061  return GO_RESULT_OK;
1062  }
