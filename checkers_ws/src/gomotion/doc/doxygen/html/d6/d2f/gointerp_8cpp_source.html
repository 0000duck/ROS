1 /*
2  gointerp.c
3 
4  Polynomial interpolation of various orders: constant, linear, cubic,
5  quintic. There are two types, point fit and boundary fit. For an
6  nth-order point fit, n+1 points are provided and the interpolating
7  polynomial fits them exactly. For a boundary fit, (n+1)/2 tuples
8  of points and derivatives are provided and the interpolating polynomial
9  fits the points and derivatives exactly.
10 
11  Time is normalized to t = 0 at start of interval of interpolation,
12  t = 1 at end.
13 
14  Point-fit interpolation is undesirable for motion control and is
15  included here to be used for comparison.
16 
17  Boundary interpolation should be used for motion control. If the
18  derivatives are not known, they can be estimated by differencing.
19  Differencing introduces slight reversal and overshoot errors at the
20  beginning and end of step motions, more pronounced the higher the degree.
21 
22  For constant- and linear interpolation, point-fit and boundary types
23  are the same.
24 */

237 /*
238  Add a point to the boundary cubic interpolator. Since you are not providing
239  the necessary velocity partner, it will be estimated by differencing the
240  positions. First the point is added by shifting the previous, as with
241  the exact-fit interpolator. Then two intermediate p-v pairs are computed
242  for the middle of the three intervals, by averaging the differenced
243  velocities at the ends of the middle interval. The boundary cubic
244  coefficients are the computed. Calling this four times from scratch will
245  fill the interpolator and make it ready to go.
246 
247  This gives better behavior than the exact fit cubic for motion control,
248  although not as good as if the p-v pairs were computed exactly, from
249  the inverse Jacobian for example.
250  */
251 go_result go_interp::add_cubic_pdv( go_real pt)
252 {
253  points p;
254  go_real v0, v1;
255  go_real wp;
256  go_real k = 5.0;
257  go_real kp2_inv = 1.0 / (k + 2.0);
258 
259  /* estimate boundary values */
260  v0 = 0.5 * (this->p.p[2] - this->p.p[0]);
261  v1 = 0.5 * (pt - this->p.p[1]);
262  /* set up point structure */
263  wp = this->p.p[1]; /* original */
264  wp = kp2_inv * (this->p.p[0] + k * this->p.p[1] + this->p.p[2]); /* blended */
265  p.p[0] = wp;
266  p.p[1] = v0;
267  wp = this->p.p[2]; /* original */
268  wp = kp2_inv * (this->p.p[1] + k * this->p.p[2] + pt); /* blended */
269  p.p[2] = wp;
270  p.p[3] = v1;
271  /* shift points down */
272  this->p.p[0] = this->p.p[1];
273  this->p.p[1] = this->p.p[2];
274  /* add new point */
275  this->p.p[2] = pt;
276  /* calculate coeff from point structure */
277  return calc_coeff_cubic_bc(&p, &this->a);
278 }
