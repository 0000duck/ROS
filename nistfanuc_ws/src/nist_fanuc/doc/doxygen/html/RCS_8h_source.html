69 inline std::vector<T> ToVector(int n, ...) {
70  std::vector<T> ds;
71  va_list args; // define argument list variable
72  va_start(args, n); // init list; point to last
73  // defined argument
74 
75  for (int i = 0; i < n; i++) {
76  double d = va_arg(args, T); // get next argument

78  }
79  va_end(args); // clean up the system stack
80  return ds;
